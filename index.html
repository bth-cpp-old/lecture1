makefile

Exempel på en makefile


							CXX=clang++
							CFLAGS=-c -std=c++11
							EXECUTABLE=program

							SRC=main.cpp # Add all .cpp here
							OBJECTS=$(SRC:.cpp=.o)

							all: $(OBJECTS)
								$(CXX) $(OBJECTS) -o $(EXECUTABLE)
								@echo "Done."

							main.o: main.cpp
								@echo "building $@"
								$(CXX) $(CFLAGS) main.cpp

							clean:
								@rm -f $(EXECUTABLE)
								@find . -name "*.o" -print0 | xargs -0 rm -f
								@echo "Cleaned files."
						

							# Syntax for rules in makefile
